rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null &&
             request.auth.token.email == 'mathieudubris@gmail.com';
    }

    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return request.auth != null && request.auth.uid == userId;
    }

    function isProjectMember(projectId) {
      return request.auth != null && 
             request.auth.uid in get(/databases/$(database)/documents/portfolio/projet-en-cours/projects/$(projectId)).data.teamMembers;
    }

    function isProjectCreator(projectId) {
      return request.auth != null && 
             request.auth.uid == get(/databases/$(database)/documents/portfolio/projet-en-cours/projects/$(projectId)).data.createdBy;
    }

    function isBoardMember(boardId) {
      return request.auth != null &&
             request.auth.uid in get(/databases/$(database)/documents/kanban_boards/$(boardId)).data.members;
    }

    function isFormationMember(formationId) {
      return request.auth != null &&
             request.auth.uid in get(/databases/$(database)/documents/services/formation/formations/$(formationId)).data.teamMembers;
    }

    function isFormationCreator(formationId) {
      return request.auth != null &&
             request.auth.uid == get(/databases/$(database)/documents/services/formation/formations/$(formationId)).data.createdBy;
    }

    function isAccompagnementMember(accompagnementId) {
      return request.auth != null &&
             request.auth.uid in get(/databases/$(database)/documents/services/accompagnement/accompagnements/$(accompagnementId)).data.teamMembers;
    }

    function isAccompagnementCreator(accompagnementId) {
      return request.auth != null &&
             request.auth.uid == get(/databases/$(database)/documents/services/accompagnement/accompagnements/$(accompagnementId)).data.createdBy;
    }

    // ─────────────────────────────────────────
    // BLOGS
    // ─────────────────────────────────────────

    match /blogs/{blogId} {
      allow read: if true;
      allow create, update, delete: if request.auth != null;

      match /comments/{commentId} {
        allow read: if true;
        allow create: if request.auth != null;
        allow update, delete: if request.auth != null;
      }

      match /likes/{userId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.uid == userId;
      }

      match /views/{userId} {
        allow read: if true;
        allow write: if request.auth != null;
      }
    }

    match /blogs/{blogId}/comments/{commentId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null &&
          (request.auth.uid == resource.data.authorId || isAdmin());
    }

    // ─────────────────────────────────────────
    // USERS
    // ─────────────────────────────────────────

    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null &&
          (request.auth.uid == userId || isAdmin());
      allow delete: if isAdmin();

      match /preferences/{prefId} {
        allow read, write: if isAuthenticated();
      }

      match /dailyLogs/{date} {
        allow read, write: if isAuthenticated();
      }

      match /bodyMetrics/{metricId} {
        allow read, write: if isAuthenticated();
      }

      match /alerts/{alertId} {
        allow read, write: if isAuthenticated();
      }

      match /savedPosts/{postId} {
        allow read, write: if isAuthenticated();
      }
    }

    // ─────────────────────────────────────────
    // TEAM
    // ─────────────────────────────────────────

    match /team/{userId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null &&
          (request.auth.uid == userId || isAdmin());
    }

    // ─────────────────────────────────────────
    // PORTFOLIO / PROJETS
    // ─────────────────────────────────────────

    match /portfolio/{document} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /portfolio/projet-en-cours/{document} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /portfolio/projet-en-cours/projects/{projectId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && isAdmin();
      allow update: if request.auth != null && (isAdmin() || request.auth.uid == resource.data.createdBy);
      allow delete: if isAdmin();

      match /{document} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /overview/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /media/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /software/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /team/{memberId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null &&
            request.auth.uid == request.resource.data.userId &&
            request.resource.data.projectId == projectId;
        allow update: if request.auth != null &&
            request.auth.uid == resource.data.userId;
        allow delete: if request.auth != null &&
            (request.auth.uid == resource.data.userId || isAdmin());
      }

      match /roadmap/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /documentation/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /stats/{docId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
      }

      match /progression/{boardId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
        allow update: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
        allow delete: if request.auth != null && (isAdmin() || isProjectCreator(projectId));

        match /kanban_columns/{columnId} {
          allow read: if request.auth != null;
          allow write: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
        }

        match /kanban_cards/{cardId} {
          allow read: if request.auth != null;
          allow create: if request.auth != null &&
              (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId)) &&
              request.resource.data.createdBy == request.auth.uid;
          allow update: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId));
          allow delete: if request.auth != null && (isAdmin() || isProjectCreator(projectId) || isProjectMember(projectId) ||
              request.auth.uid == resource.data.createdBy);
        }
      }
    }

    // ─────────────────────────────────────────
    // PROJECT INVITATIONS
    // ─────────────────────────────────────────

    match /project_invitations/{invitationId} {
      allow read: if request.auth != null && (
        isAdmin() ||
        request.auth.email == resource.data.email ||
        request.auth.uid == resource.data.invitedBy
      );
      allow create: if isAdmin() &&
          request.resource.data.projectId is string &&
          request.resource.data.email is string &&
          request.resource.data.email.matches('^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$') &&
          request.resource.data.status == 'pending';
      allow update: if request.auth != null && (
        (request.auth.email == resource.data.email &&
         request.resource.data.status == 'accepted') ||
        isAdmin()
      );
      allow delete: if isAdmin();
    }

    // ─────────────────────────────────────────
    // LICENCES
    // ─────────────────────────────────────────

    match /licences/{licenceId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // ─────────────────────────────────────────
    // VIDEOS
    // ─────────────────────────────────────────

    match /videos/{videoId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null &&
          (request.auth.uid == resource.data.userId || isAdmin());

      match /likes/{userId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.uid == userId;
      }

      match /views/{userId} {
        allow read: if true;
        allow write: if request.auth != null;
      }
    }

    // ─────────────────────────────────────────
    // NOUVEAUTÉS
    // ─────────────────────────────────────────

    match /nouveautes/{nouveauteId} {
      allow read: if true;
      allow create: if isAdmin() &&
          request.resource.data.sourceId is string &&
          request.resource.data.type in ['project', 'blog'] &&
          request.resource.data.position is number &&
          request.resource.data.position >= 1 &&
          request.resource.data.position <= 5 &&
          request.resource.data.title is string &&
          request.resource.data.link is string;
      allow update: if isAdmin() &&
          request.resource.data.position is number &&
          request.resource.data.position >= 1 &&
          request.resource.data.position <= 5;
      allow delete: if isAdmin();
    }

    // ─────────────────────────────────────────
    // FINANCE
    // ─────────────────────────────────────────

    match /finance_transactions/{txId} {
      allow read: if request.auth != null &&
                  (request.auth.uid == resource.data.userId || isAdmin());
      allow create: if request.auth != null &&
                    request.auth.uid == request.resource.data.userId &&
                    request.resource.data.amount is number &&
                    request.resource.data.amount > 0 &&
                    request.resource.data.type in ['income', 'expense', 'saving'] &&
                    request.resource.data.label is string &&
                    request.resource.data.label.size() > 0 &&
                    request.resource.data.currency in ['EUR', 'USD', 'GBP'];
      allow update: if request.auth != null &&
                    (request.auth.uid == resource.data.userId || isAdmin()) &&
                    request.resource.data.userId == resource.data.userId;
      allow delete: if request.auth != null &&
                    (request.auth.uid == resource.data.userId || isAdmin());
    }

    // ─────────────────────────────────────────
    // KANBAN (boards indépendants)
    // ─────────────────────────────────────────

    match /kanban_boards/{boardId} {
      allow read: if request.auth != null &&
          (request.auth.uid in resource.data.members || isAdmin());
      allow create: if request.auth != null &&
          request.resource.data.createdBy == request.auth.uid &&
          request.auth.uid in request.resource.data.members &&
          request.resource.data.title is string &&
          request.resource.data.title.size() > 0;
      allow update: if request.auth != null &&
          (request.auth.uid in resource.data.members || isAdmin());
      allow delete: if request.auth != null &&
          (request.auth.uid == resource.data.createdBy || isAdmin());
    }

    match /kanban_columns/{columnId} {
      allow read: if request.auth != null &&
          isBoardMember(resource.data.boardId);
      allow create: if request.auth != null &&
          request.resource.data.boardId is string &&
          request.resource.data.title is string &&
          request.resource.data.position is number &&
          isBoardMember(request.resource.data.boardId);
      allow update: if request.auth != null &&
          isBoardMember(resource.data.boardId);
      allow delete: if request.auth != null &&
          isBoardMember(resource.data.boardId);
    }

    match /kanban_cards/{cardId} {
      allow read: if request.auth != null &&
          isBoardMember(resource.data.boardId);
      allow create: if request.auth != null &&
          request.resource.data.boardId is string &&
          request.resource.data.columnId is string &&
          request.resource.data.title is string &&
          request.resource.data.position is number &&
          request.resource.data.createdBy == request.auth.uid &&
          isBoardMember(request.resource.data.boardId);
      allow update: if request.auth != null &&
          isBoardMember(resource.data.boardId);
      allow delete: if request.auth != null &&
          (request.auth.uid == resource.data.createdBy || isBoardMember(resource.data.boardId) || isAdmin());
    }

    // ─────────────────────────────────────────
    // SERVICES > FORMATION
    // Structure : services/formation/formations/{formationId}
    // ─────────────────────────────────────────

    match /services/formation/formations/{formationId} {
      allow read: if request.auth != null && (
        isAdmin() ||
        request.auth.uid == resource.data.createdBy ||
        request.auth.uid in resource.data.teamMembers
      );
      allow create: if isAdmin() &&
          request.resource.data.title is string &&
          request.resource.data.title.size() > 0 &&
          request.resource.data.slug is string &&
          request.resource.data.createdBy == request.auth.uid &&
          request.resource.data.visibility in ['public', 'members_only'];
      allow update: if isAdmin();
      allow delete: if isAdmin();

      match /overview/{docId} {
        allow read: if request.auth != null && (isAdmin() || isFormationCreator(formationId) || isFormationMember(formationId));
        allow write: if isAdmin();
      }

      match /media/{docId} {
        allow read: if request.auth != null && (isAdmin() || isFormationCreator(formationId) || isFormationMember(formationId));
        allow write: if isAdmin();
      }

      match /content/{docId} {
        allow read: if request.auth != null && (isAdmin() || isFormationCreator(formationId) || isFormationMember(formationId));
        allow write: if isAdmin();
      }

      match /stats/{docId} {
        allow read: if request.auth != null && (isAdmin() || isFormationCreator(formationId) || isFormationMember(formationId));
        allow update: if request.auth != null && (isAdmin() || isFormationCreator(formationId) || isFormationMember(formationId));
        allow create: if isAdmin();
        allow delete: if isAdmin();
      }
    }

    // ─────────────────────────────────────────
    // SERVICES > ACCOMPAGNEMENT
    // Structure : services/accompagnement/accompagnements/{accompagnementId}
    // ─────────────────────────────────────────

    match /services/{serviceDoc} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /services/accompagnement/accompagnements/{accompagnementId} {
      // Lecture : admin, créateur, ou bénéficiaire autorisé
      allow read: if request.auth != null && (
        isAdmin() ||
        request.auth.uid == resource.data.createdBy ||
        request.auth.uid in resource.data.teamMembers
      );
      // Création : admin uniquement
      allow create: if isAdmin() &&
          request.resource.data.title is string &&
          request.resource.data.title.size() > 0 &&
          request.resource.data.slug is string &&
          request.resource.data.createdBy == request.auth.uid &&
          request.resource.data.visibility in ['public', 'members_only'];
      // Mise à jour : admin uniquement
      allow update: if isAdmin();
      // Suppression : admin uniquement
      allow delete: if isAdmin();

      // Sous-collection overview
      match /overview/{docId} {
        allow read: if request.auth != null && (
          isAdmin() ||
          isAccompagnementCreator(accompagnementId) ||
          isAccompagnementMember(accompagnementId)
        );
        allow write: if isAdmin();
      }

      // Sous-collection media
      match /media/{docId} {
        allow read: if request.auth != null && (
          isAdmin() ||
          isAccompagnementCreator(accompagnementId) ||
          isAccompagnementMember(accompagnementId)
        );
        allow write: if isAdmin();
      }

      // Sous-collection content (étapes)
      match /content/{docId} {
        allow read: if request.auth != null && (
          isAdmin() ||
          isAccompagnementCreator(accompagnementId) ||
          isAccompagnementMember(accompagnementId)
        );
        allow write: if isAdmin();
      }

      // Sous-collection stats (vues)
      match /stats/{docId} {
        allow read: if request.auth != null && (
          isAdmin() ||
          isAccompagnementCreator(accompagnementId) ||
          isAccompagnementMember(accompagnementId)
        );
        allow update: if request.auth != null && (
          isAdmin() ||
          isAccompagnementCreator(accompagnementId) ||
          isAccompagnementMember(accompagnementId)
        );
        allow create: if isAdmin();
        allow delete: if isAdmin();
      }
    }

  }
}
