// pages/robots.js
function Robots() {}

export async function getServerSideProps({ res }) {
  const robots = `User-agent: *
Allow: /
Sitemap: https://mathieu-dubris.web.app/sitemap.xml

# Règles supplémentaires pour Google
User-agent: Googlebot
Allow: /
Crawl-delay: 1`;

  res.setHeader('Content-Type', 'text/plain');
  res.write(robots);
  res.end();

  return {
    props: {},
  };
}

export default Robots;
